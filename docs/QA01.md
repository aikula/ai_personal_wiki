## 1. `token_budget.py` — сигнатуры и интерфейс

```python
# app/core/token_budget.py

class CharLimitError(Exception):
    """Raised when a single unit exceeds absolute hard limit."""
    pass


class ContextBudget:
    """
    Manages character budgets for different context slots.
    Uses chars (not tokens) for speed — 1 token ≈ 4 chars is sufficient heuristic.
    """

    # Default budgets (chars) — come from settings.yaml
    agents_md:      int = 4_200
    skills_md:      int = 8_750
    index_l0:       int = 10_500
    index_l1:       int = 5_250
    entity_page:    int = 3_500
    concept_page:   int = 5_250
    log_md:         int = 3_500
    conflicts_md:   int = 35_000
    wiki_context:   int = 21_000   # total for all pages in one query
    history:        int = 7_000    # chat history budget

    def __init__(self, settings: "Settings | None" = None):
        if settings:
            lim = settings.limits
            self.agents_md   = lim.agents_md_chars
            self.skills_md   = lim.skills_md_chars
            self.wiki_context = settings.query.context_budget_chars
            self.history     = settings.query.history_budget_chars
            # ... остальные из lim.*

    def count_chars(self, text: str) -> int:
        """Return character count. Simple len()."""
        return len(text)

    def trim(self, text: str, slot: str) -> str:
        """
        Trim text to slot budget.
        slot: "wiki_context" | "agents_md" | "skills_md" | ...
        Trims from the end, adds truncation marker.
        """
        limit: int = getattr(self, slot, self.wiki_context)
        if len(text) <= limit:
            return text
        return text[:limit - 40] + "\n\n… [TRIMMED — char limit reached]"

    def fit_wiki_pages(self, pages: list[str]) -> list[str]:
        """
        Fit as many wiki pages as possible into wiki_context budget.
        Returns subset of pages that fits, trimming last one if needed.
        Maintains page order (relevance ranking preserved).
        """
        result = []
        used = 0
        for page in pages:
            page_len = len(page)
            if used + page_len <= self.wiki_context:
                result.append(page)
                used += page_len
            else:
                remaining = self.wiki_context - used
                if remaining > 300:  # worth including truncated version
                    result.append(page[:remaining] + "\n…[TRIMMED]")
                break
        return result

    def check(self, text: str, slot: str, raise_on_exceed: bool = False) -> bool:
        """
        Check if text fits slot budget.
        If raise_on_exceed=True → raises CharLimitError.
        Returns True if fits, False if exceeds.
        """
        limit: int = getattr(self, slot, self.wiki_context)
        fits = len(text) <= limit
        if not fits and raise_on_exceed:
            raise CharLimitError(
                f"Text ({len(text)} chars) exceeds {slot} limit ({limit} chars)"
            )
        return fits
```

**Вывод:** `ContextBudget`, `count_chars`, `CharLimitError` — подтверждены. Дополнительно нужен `fit_wiki_pages` и `check`.

## 2. `interpreter.py` — интерфейс `CodeInterpreter.execute()`

**Ответ: subprocess-изоляция, не `exec()`.** `exec()` не подходит — опасно при wiki-операциях с файловой системой и не изолирует глобальный state.

```python
# app/core/interpreter.py

import subprocess
import sys
import textwrap
import tempfile
import json
from pathlib import Path
from dataclasses import dataclass


@dataclass
class ExecutionResult:
    stdout: str
    stderr: str
    exit_code: int
    success: bool
    result_json: dict | None  # если код вывел JSON в stdout

    def to_dict(self) -> dict:
        return {
            "stdout": self.stdout[:2000],
            "stderr": self.stderr[:500],
            "success": self.success,
            "result": self.result_json,
        }


class CodeInterpreter:
    """
    Executes Python code snippets in isolated subprocess.
    wiki_root is injected as WIKI_ROOT env var and available in code.

    Input:  str (Python code)
    Output: ExecutionResult

    Timeout: 15 seconds hard limit.
    stdout captured, stderr captured separately.
    Code can print JSON to stdout → parsed into result_json.
    """

    def __init__(self, wiki_root: Path, timeout: int = 15):
        self.wiki_root = wiki_root
        self.timeout = timeout

    def execute(self, code: str) -> ExecutionResult:
        """
        Run code in subprocess.
        Injected globals available in the script:
          WIKI_ROOT = str(self.wiki_root)

        Example code:
            import os, json
            files = os.listdir(WIKI_ROOT + "/wiki")
            print(json.dumps({"count": len(files)}))
        """
        # Prepend wiki_root injection
        preamble = f'WIKI_ROOT = {str(self.wiki_root)!r}\n'
        full_code = preamble + textwrap.dedent(code)

        with tempfile.NamedTemporaryFile(
            mode="w", suffix=".py", delete=False, encoding="utf-8"
        ) as f:
            f.write(full_code)
            script_path = f.name

        try:
            proc = subprocess.run(
                [sys.executable, script_path],
                capture_output=True,
                text=True,
                timeout=self.timeout,
            )
            stdout = proc.stdout.strip()
            stderr = proc.stderr.strip()
            success = proc.returncode == 0

            # Try parse JSON from stdout
            result_json = None
            if stdout:
                try:
                    result_json = json.loads(stdout)
                except json.JSONDecodeError:
                    pass

            return ExecutionResult(
                stdout=stdout,
                stderr=stderr,
                exit_code=proc.returncode,
                success=success,
                result_json=result_json,
            )
        except subprocess.TimeoutExpired:
            return ExecutionResult(
                stdout="", stderr="Execution timeout",
                exit_code=-1, success=False, result_json=None,
            )
        finally:
            Path(script_path).unlink(missing_ok=True)
```

**Итог:** `execute(code: str) → ExecutionResult`. subprocess + tempfile. `WIKI_ROOT` инжектируется как строковая переменная в начало кода. JSON-вывод из stdout автоматически парсится.

```python
# Вставить в класс WikiFS

def _render_conflict_block(self, entry: dict) -> str:
    """
    Render a ConflictEntry dict → markdown block for conflicts.md.
    entry keys: id, status, date, project, source_file, conflict_type,
                page_a_slug, page_b_ref, context_a, context_b,
                suggested_options, user_comment
    """
    options_text = "\n".join(
        f"  {i+1}. {opt}"
        for i, opt in enumerate(entry.get("suggested_options", []))
    )
    return f"""## [{entry['status']}] {entry['id']}

- **Date:** {entry['date']}
- **Project:** {entry['project']}
- **Source file:** {entry['source_file']}
- **Conflict type:** {entry['conflict_type']}
- **Page A:** [[{entry['page_a_slug']}]]
- **Page B:** {entry['page_b_ref']}
- **Context A:** {entry.get('context_a', '')}
- **Context B:** {entry.get('context_b', '')}
- **Suggested options:**
{options_text}
- **User comment:** {entry.get('user_comment', '_none_')}
- **Resolution:** {entry.get('resolution', 'pending')}
- **Skill extracted:** {entry.get('skill_extracted', '')}
- **Resolved at:** {entry.get('resolved_at', '')}

---
"""


def _render_log_entry(self, entry: dict) -> str:
    """
    Render a log event dict → single markdown list item for log.md.
    entry keys: timestamp, event_type, source_file, project,
                pages_created, pages_updated, conflict_ids, notes
    """
    pages_c = ", ".join(entry.get("pages_created", [])) or "—"
    pages_u = ", ".join(entry.get("pages_updated", [])) or "—"
    conflicts = ", ".join(entry.get("conflict_ids", [])) or "—"
    return (
        f"- **[{entry['timestamp']}]** `{entry['event_type']}` "
        f"| project: `{entry['project']}` "
        f"| file: `{entry.get('source_file', '—')}` "
        f"| created: {pages_c} "
        f"| updated: {pages_u} "
        f"| conflicts: {conflicts}"
        + (f" | {entry['notes']}" if entry.get("notes") else "")
        + "\n"
    )


def _inject_after_conflict_id(
    self,
    raw: str,
    conflict_id: str,
    text_to_inject: str,
) -> str:
    """
    Inject text_to_inject after the header line of conflict_id block.
    Used for adding comments to existing conflict entries.

    raw:              full content of conflicts.md
    conflict_id:      e.g. "CONFLICT-007"
    text_to_inject:   line(s) to insert after the ## [...] CONFLICT-NNN header
    """
    import re
    pattern = rf"(## \[(?:OPEN|RESOLVED)\] {re.escape(conflict_id)}[^\n]*\n)"
    replacement = rf"\1{text_to_inject}\n"
    updated = re.sub(pattern, replacement, raw, count=1)
    if updated == raw:
        # Fallback: append to end of file if pattern not found
        updated = raw.rstrip() + f"\n\n{text_to_inject}\n"
    return updated
```

## 7. `_parse_json_block`, `_extract_wikilinks`, `_now_iso` — общий модуль

**Выносить в `app/core/utils.py`.** Дублирование между `query_agent.py` и `audit_agent.py` — это неправильно, при изменении логики потребуется менять в двух местах.

```python
# app/core/utils.py

import json
import re
from datetime import datetime


def parse_json_block(text: str) -> dict:
    """
    Extract and parse JSON from LLM response.
    Handles ```json ... ``` fences and bare JSON objects.
    Raises ValueError if no valid JSON found.
    """
    # Try fenced block first
    fence = re.search(r"```(?:json)?\s*(\{.*?\}|\[.*?\])\s*```", text, re.DOTALL)
    if fence:
        return json.loads(fence.group(1))

    # Try bare JSON object
    obj = re.search(r"\{.*\}", text, re.DOTALL)
    if obj:
        return json.loads(obj.group(0))

    # Try bare JSON array
    arr = re.search(r"\[.*\]", text, re.DOTALL)
    if arr:
        return json.loads(arr.group(0))

    raise ValueError(f"No valid JSON found in LLM response: {text[:200]!r}")


def extract_wikilinks(text: str) -> list[str]:
    """
    Extract all [[slug]] and [[slug|text]] references from text.
    Returns list of unique slugs (without display text or anchors).
    """
    raw = re.findall(r"\[\[([^\]]+)\]\]", text)
    slugs = []
    seen = set()
    for item in raw:
        # Strip display text [[slug|text]] → slug
        slug = item.split("|")[0].strip()
        # Strip anchor [[slug#anchor]] → slug
        slug = slug.split("#")[0].strip()
        if slug and slug not in seen:
            slugs.append(slug)
            seen.add(slug)
    return slugs


def now_iso() -> str:
    """Current datetime as ISO-8601 string, seconds precision."""
    return datetime.now().isoformat(timespec="seconds")
```

**В `query_agent.py` и `audit_agent.py` заменить:**
```python
# Было (локальные определения):
def _parse_json_block(text): ...
def _extract_wikilinks(text): ...
def _now_iso(): ...

# Стало (один импорт):
from app.core.utils import parse_json_block, extract_wikilinks, now_iso

# Внутри методов:
_parse_json_block(raw)   → parse_json_block(raw)
_extract_wikilinks(text) → extract_wikilinks(text)
_now_iso()               → now_iso()
```
